/* Crea la lista de variables de entorno a partir de **envp y devuelve el 
puntero al head de la lista */
// OLD en env_init.c
static t_env	*env_copy_envp(char **envp)
{
	t_env	*node;
	t_env	*head;
	t_env	*prev;
	int		i;

	i = 0;
	prev = NULL;
	while (envp[i])
	{
		node = env_new_node();
		if (!node)
			exit(EXIT_FAILURE);
		if (i == 0)
			head = node;
		if (env_get_content(envp[i], node))
			exit(EXIT_FAILURE);
		env_add_node(&prev, node);
		i++;
	}
	return (head);
}
